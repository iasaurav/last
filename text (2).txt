function ytab() {
  var sheet = SpreadsheetApp.getActive().getSheetByName("Sheet2")
  var lastRow = sheet.getLastRow();
  var formula = '=if(A2="","", youtube(B2,A2))';
  var sourceRange = sheet.getRange(2, 3);
  sourceRange.setFormula(formula);
  var destinationRange = sheet.getRange(2, 3, lastRow - 1, 1);
  sourceRange.autoFill(destinationRange, SpreadsheetApp.
    AutoFillSeries.DEFAULT_SERIES);
}
